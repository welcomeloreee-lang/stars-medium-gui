local function SavePresets()
	writefile("EmotePresets.txt", HttpService:JSONEncode(Presets))
end

local function LoadPresets()
	if isfile("EmotePresets.txt") then
		pcall(function()
			Presets = HttpService:JSONDecode(readfile("EmotePresets.txt"))
		end)
	end
	
	for i = 1, 10 do
		if not Presets[i] then
			Presets[i] = {}
		end
	end
end

local function SaveCustomOrder()
	local customOrder = {}
	for i, emote in ipairs(Emotes) do
		customOrder[emote.id] = emote.sort.customorder
	end
	writefile("EmoteCustomOrder.txt", HttpService:JSONEncode(customOrder))
end

local function LoadCustomOrder()
	if isfile("EmoteCustomOrder.txt") then
		pcall(function()
			local customOrder = HttpService:JSONDecode(readfile("EmoteCustomOrder.txt"))
			for _, emote in ipairs(Emotes) do
				if customOrder[emote.id] then
					emote.sort.customorder = customOrder[emote.id]
				end
			end
		end)
	end
end

local function LoadPreset(presetNum)
	CurrentPreset = presetNum
	FavoritedEmotes = {}
	for _, id in ipairs(Presets[presetNum]) do
		table.insert(FavoritedEmotes, id)
	end
	writefile("FavoritedEmotes.txt", HttpService:JSONEncode(FavoritedEmotes))
	
	
	for _, EmoteButton in pairs(Frame:GetChildren()) do
		if EmoteButton:IsA("ImageButton") or EmoteButton:IsA("TextButton") then
			local emoteId = tonumber(EmoteButton.Name)
			if emoteId then
				local Favorite = EmoteButton:FindFirstChild("favorite")
				if Favorite then
					local IsFavorited = table.find(FavoritedEmotes, emoteId)
					Favorite.Image = IsFavorited and FavoriteOn or FavoriteOff
					
					-- Update layout order
					for _, Emote in pairs(Emotes) do
						if Emote.id == emoteId then
							EmoteButton.LayoutOrder = Emote.sort[CurrentSort] + ((IsFavorited and 0) or #Emotes)
							break
						end
					end
				end
			end
		end
	end
end

local function SaveCurrentToPreset(presetNum)
	Presets[presetNum] = {}
	for _, id in ipairs(FavoritedEmotes) do
		table.insert(Presets[presetNum], id)
	end
	SavePresets()
end

local function SortEmotes()
	for i, Emote in pairs(Emotes) do
		local EmoteButton = Frame:FindFirstChild(tostring(Emote.id))
		if EmoteButton then
			local IsFavorited = table.find(FavoritedEmotes, Emote.id)
			EmoteButton.LayoutOrder = Emote.sort[CurrentSort] + ((IsFavorited and 0) or #Emotes)
			local numberLabel = EmoteButton:FindFirstChild("number")
			if numberLabel then
				numberLabel.Text = tostring(Emote.sort[CurrentSort])
			end
		end
	end
end

-- NEW: Function to toggle favorites visibility
local function ToggleFavorites()
	FavoritesEnabled = not FavoritesEnabled
	FavoritesToggle.Text = FavoritesEnabled and "‚≠ê: On" or "‚≠ê: Off"
	
	-- Update visibility of all favorite buttons
	for _, EmoteButton in pairs(Frame:GetChildren()) do
		if EmoteButton:IsA("ImageButton") then
			local Favorite = EmoteButton:FindFirstChild("favorite")
			if Favorite then
				Favorite.Visible = FavoritesEnabled
			end
		end
	end
end


local function ToggleCustomDragMode()
	CustomDragMode = not CustomDragMode
	
	if CustomDragMode then
		CustomButton.BackgroundColor3 = Color3.fromRGB(80, 120, 200)
		CustomButton.Text = "Custom ‚úì"
		-- Switch to custom order if not already
		if CurrentSort ~= "customorder" then
			CurrentSort = "customorder"
			SortEmotes()
		end
	else
		CustomButton.BackgroundColor3 = Color3.fromRGB(60, 65, 70)
		CustomButton.Text = "Custom"
	end
end


local function SwapEmotes(emote1Id, emote2Id)
	local emote1, emote2
	
	for _, emote in ipairs(Emotes) do
		if emote.id == emote1Id then
			emote1 = emote
		elseif emote.id == emote2Id then
			emote2 = emote
		end
	end
	
	if emote1 and emote2 then
		-- Swap their custom order values
		local temp = emote1.sort.customorder
		emote1.sort.customorder = emote2.sort.customorder
		emote2.sort.customorder = temp
		
		-- Update UI
		SortEmotes()
		SaveCustomOrder()
	end
end

		if CurrentPreset ~= presetNum then
			loadBtn.BackgroundColor3 = Color3.fromRGB(80, 85, 90)
		end
	end)

	loadBtn.MouseLeave:Connect(function()
		loadBtn.BackgroundColor3 = CurrentPreset == presetNum and Color3.fromRGB(80, 120, 200) or Color3.fromRGB(60, 65, 70)
	end)

	loadBtn.MouseButton1Click:Connect(function()
		LoadPreset(presetNum)
		PresetsFrame.Visible = false
		-- Update all preset buttons
		for _, child in pairs(PresetsFrame:GetChildren()) do
			if child:IsA("Frame") then
				local btn = child:FindFirstChildOfClass("TextButton")
				if btn and btn.Text:match("Preset") then
					local num = tonumber(btn.Text:match("%d+"))
					btn.BackgroundColor3 = num == presetNum and Color3.fromRGB(80, 120, 200) or Color3.fromRGB(60, 65, 70)
				end
			end
		end
	end)

	saveBtn.MouseEnter:Connect(function()
		saveBtn.BackgroundColor3 = Color3.fromRGB(100, 160, 100)
	end)

	saveBtn.MouseLeave:Connect(function()
		saveBtn.BackgroundColor3 = Color3.fromRGB(80, 140, 80)
	end)

	saveBtn.MouseButton1Click:Connect(function()
		SaveCurrentToPreset(presetNum)
		-- Visual feedback
		saveBtn.Text = "Saved!"
		wait(0.5)
		saveBtn.Text = "Save"
	end)

	return container
end

local function SendNotification(title, text)
	if (not IsStudio) and syn and syn.toast_notification then
		syn.toast_notification({ Type = ToastType.Error, Title = title, Content = text })
	else
		StarterGui:SetCore("SendNotification", { Title = title, Text = text })
	end
end

local function HumanoidPlayEmote(humanoid, name, id)
	if IsStudio then
		return humanoid:PlayEmote(name)
	else
		return humanoid:PlayEmoteAndGetAnimTrackById(id)
	end
end

local function PlayEmote(name, id)
	SearchBar.Text = ""
	local Humanoid = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
	local Description = Humanoid and Humanoid:FindFirstChildOfClass("HumanoidDescription")
	if not (Humanoid and Description) then return end
	if LocalPlayer.Character.Humanoid.RigType ~= Enum.HumanoidRigType.R6 then
		local succ = pcall(function()
			HumanoidPlayEmote(Humanoid, name, id)
		end)
		if not succ then
			Description:AddEmote(name, id)
			HumanoidPlayEmote(Humanoid, name, id)
		end
	else
		SendNotification("R6 Not Supported", "Please use R15 to play emotes")
	end
end

local function WaitForChildOfClass(parent, class)
	local child = parent:FindFirstChildOfClass(class)
	while not child or child.ClassName ~= class do
		child = parent.ChildAdded:Wait()
	end
	return child
end

-- Character setup
local function CharacterAdded(Character)
	for _, v in pairs(Frame:GetChildren()) do
		if not v:IsA("UIGridLayout") then
			v:Destroy()
		end
	end

	local Humanoid = WaitForChildOfClass(Character, "Humanoid")
	local Description = Humanoid:WaitForChild("HumanoidDescription", 5) or Instance.new("HumanoidDescription", Humanoid)


	local random = Instance.new("TextButton")
	random.Size = UDim2.new(0, 87, 0, 80)
	random.LayoutOrder = 0
	random.TextColor3 = Color3.new(1, 1, 1)
	random.BorderSizePixel = 0
	random.BackgroundColor3 = Color3.fromRGB(100, 120, 100)
	random.BackgroundTransparency = 0.4
	random.Font = Enum.Font.GothamBold
	random.TextSize = 16
	random.Text = "Random"
	random:SetAttribute("name", "Random")

	local randomCorner = Instance.new("UICorner")
	randomCorner.CornerRadius = UDim.new(0, 8)
	randomCorner.Parent = random

	random.MouseButton1Click:Connect(function()
		local randomemote = Emotes[math.random(1, #Emotes)]
		PlayEmote(randomemote.name, randomemote.id)
	end)

	random.MouseEnter:Connect(function()
		EmoteName.Text = "Random Emote"
		random.BackgroundTransparency = 0.2
	end)

	random.MouseLeave:Connect(function()
		random.BackgroundTransparency = 0.4
	end)

	random.Parent = Frame


	for i, Emote in pairs(Emotes) do
		Description:AddEmote(Emote.name, Emote.id)

		local EmoteButton = Instance.new("ImageButton")
		local IsFavorited = table.find(FavoritedEmotes, Emote.id)

		EmoteButton.LayoutOrder = Emote.sort[CurrentSort] + ((IsFavorited and 0) or #Emotes)
		EmoteButton.Name = tostring(Emote.id)
		EmoteButton:SetAttribute("name", Emote.name)
		EmoteButton.Size = UDim2.new(0, 87, 0, 80)
		EmoteButton.Image = Emote.icon
		EmoteButton.BackgroundColor3 = Color3.fromRGB(55, 60, 65)
		EmoteButton.BackgroundTransparency = 0.4
		EmoteButton.BorderSizePixel = 0

		local emoteCorner = Instance.new("UICorner")
		emoteCorner.CornerRadius = UDim.new(0, 8)
		emoteCorner.Parent = EmoteButton


		local EmoteNumber = Instance.new("TextLabel")
		EmoteNumber.Name = "number"
		EmoteNumber.Font = Enum.Font.GothamBold
		EmoteNumber.TextSize = 12
		EmoteNumber.BackgroundTransparency = 1
		EmoteNumber.TextColor3 = Color3.new(1, 1, 1)
		EmoteNumber.BorderSizePixel = 0
		EmoteNumber.Size = UDim2.new(0, 20, 0, 20)
		EmoteNumber.Position = UDim2.new(0, 3, 1, -23)
		EmoteNumber.Text = tostring(Emote.sort[CurrentSort])
		EmoteNumber.TextXAlignment = Enum.TextXAlignment.Center
		EmoteNumber.TextYAlignment = Enum.TextYAlignment.Center

		local UIStroke = Instance.new("UIStroke")
		UIStroke.Thickness = 2
		UIStroke.Color = Color3.fromRGB(0, 0, 0)
		UIStroke.Parent = EmoteNumber

		EmoteNumber.Parent = EmoteButton

		
		local Favorite = Instance.new("ImageButton")
		Favorite.Name = "favorite"
		Favorite.Image = IsFavorited and FavoriteOn or FavoriteOff
		Favorite.Size = UDim2.new(0, 20, 0, 20)
		Favorite.Position = UDim2.new(1, -23, 1, -23)
		Favorite.BorderSizePixel = 0
		Favorite.BackgroundTransparency = 1
		Favorite.ZIndex = 2
		Favorite.Visible = FavoritesEnabled
		Favorite.Parent = EmoteButton

		
		EmoteButton.MouseButton1Click:Connect(function()
			
			if CustomDragMode and CurrentSort == "customorder" then
				if DraggedEmote == nil then
					
					DraggedEmote = Emote.id
					EmoteButton.BackgroundColor3 = Color3.fromRGB(100, 140, 220)
					EmoteButton.BackgroundTransparency = 0.2
				else
					
					if DraggedEmote ~= Emote.id then
						SwapEmotes(DraggedEmote, Emote.id)
					end
					
					local draggedButton = Frame:FindFirstChild(tostring(DraggedEmote))
					if draggedButton then
						draggedButton.BackgroundColor3 = Color3.fromRGB(55, 60, 65)
						draggedButton.BackgroundTransparency = 0.4
					end
					DraggedEmote = nil
				end
			else
				
				PlayEmote(Emote.name, Emote.id)
			end
		end)

		EmoteButton.MouseEnter:Connect(function()
			EmoteName.Text = Emote.name
			if not (CustomDragMode and DraggedEmote == Emote.id) then
				EmoteButton.BackgroundTransparency = 0.2
			end
		end)

		EmoteButton.MouseLeave:Connect(function()
			if not (CustomDragMode and DraggedEmote == Emote.id) then
				EmoteButton.BackgroundTransparency = 0.4
			end
		end)

		Favorite.MouseButton1Click:Connect(function()
			local index = table.find(FavoritedEmotes, Emote.id)
			if index then
				table.remove(FavoritedEmotes, index)
				Favorite.Image = FavoriteOff
				EmoteButton.LayoutOrder = Emote.sort[CurrentSort] + #Emotes
			else
				table.insert(FavoritedEmotes, Emote.id)
				Favorite.Image = FavoriteOn
				EmoteButton.LayoutOrder = Emote.sort[CurrentSort]
			end
			writefile("FavoritedEmotes.txt", HttpService:JSONEncode(FavoritedEmotes))
		end)

		EmoteButton.Parent = Frame
	end
end
